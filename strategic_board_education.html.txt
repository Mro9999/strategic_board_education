        function updateAllScores() {
            // トレンド適合性
            const onlineEducation = parseFloat(document.getElementById('onlineEducation').value);
            const personalization = parseFloat(document.getElementById('personalization').value);
            const reskilling = parseFloat(document.getElementById('reskilling').value);
            const lifelongLearning = parseFloat(document.getElementById('lifelongLearning').value);
            const trendScore = (onlineEducation + personalization + reskilling + lifelongLearning) / 4;

            // ブランド力
            const institutionalTrust = parseFloat(document.getElementById('institutionalTrust').value);
            const teachingExpertise = parseFloat(document.getElementById('teachingExpertise').value);
            const studentSatisfaction = parseFloat(document.getElementById('studentSatisfaction').value);
            const industryReputation = parseFloat(document.getElementById('industryReputation').value);
            const brandScore = (institutionalTrust + teachingExpertise + studentSatisfaction + industryReputation) / 4;

            // 教育コンテンツ・サービス力
            const curriculumQuality = parseFloat(document.getElementById('curriculumQuality').value);
            const instructorQuality = parseFloat(document.getElementById('instructorQuality').value);
            const learningOutcomes = parseFloat(document.getElementById('learningOutcomes').value);
            const customization = parseFloat(document.getElementById('customization').value);
            const contentScore = (curriculumQuality + instructorQuality + learningOutcomes + customization) / 4;

            // 経営環境
            const facilityAdvantage = parseFloat(document.getElementById('facilityAdvantage').value);
            const hybridDelivery = parseFloat(document.getElementById('hybridDelivery').value);
            const digitalPlatform = parseFloat(document.getElementById('digitalPlatform').value);
            const operationalEfficiency = parseFloat(document.getElementById('operationalEfficiency').value);
            const environmentScore = (facilityAdvantage + hybridDelivery + digitalPlatform + operationalEfficiency) / 4;

            // 価格戦略
            const priceCompetitiveness = parseFloat(document.getElementById('priceCompetitiveness').value);
            const educationalROI = parseFloat(document.getElementById('educationalROI').value);
            const pricingFlexibility = parseFloat(document.getElementById('pricingFlexibility').value);
            const paymentSupport = parseFloat(document.getElementById('paymentSupport').value);
            const pricingScore = (priceCompetitiveness + educationalROI + pricingFlexibility + paymentSupport) / 4;

            // コスト効率
            const instructorCostEfficiency = parseFloat(document.getElementById('instructorCostEfficiency').value);
            const contentProductionEfficiency = parseFloat(document.getElementById('contentProductionEfficiency').value);
            const facilityCostControl = parseFloat(document.getElementById('facilityCostControl').value);
            const marketingEfficiency = parseFloat(document.getElementById('marketingEfficiency').value);
            const costScore = (instructorCostEfficiency + contentProductionEfficiency + facilityCostControl + marketingEfficiency) / 4;

            // 総合スコア計算（重み付け - 教育・研修業界特化）
            const totalScore = (
                trendScore * 0.25 +        // トレンド適合性: 25% (教育業界では重要)
                brandScore * 0.20 +        // ブランド力: 20%
                contentScore * 0.30 +      // 教育コンテンツ・サービス力: 30% (最重要)
                environmentScore * 0.15 +  // 経営環境: 15%
                pricingScore * 0.05 +      // 価格戦略: 5%
                costScore * 0.05           // コスト効率: 5%
            );

            // 表示更新
            document.getElementById('trendScore').textContent = trendScore.toFixed(1) + '%';
            document.getElementById('brandScore').textContent = brandScore.toFixed(1) + '%';
            document.getElementById('contentScore').textContent = contentScore.toFixed(1) + '%';
            document.getElementById('environmentScore').textContent = environmentScore.toFixed(1) + '%';
            document.getElementById('pricingScore').textContent = pricingScore.toFixed(1) + '%';
            document.getElementById('costScore').textContent = costScore.toFixed(1) + '%';
            document.getElementById('successRate').textContent = Math.round(totalScore) + '%';

            // チャート更新
            successChart.data.datasets[0].data = [totalScore, 100 - totalScore];
            successChart.update();

            // インサイト更新
            updateInsights(trendScore, brandScore, contentScore, environmentScore, pricingScore, costScore, totalScore);
        }

        function updateInsights(trend, brand, content, environment, pricing, cost, total) {
            const scores = [
                {name: 'トレンド適合性', score: trend},
                {name: 'ブランド力', score: brand},
                {name: '教育コンテンツ・サービス力', score: content},
                {name: '経営環境', score: environment},
                {name: '価格戦略', score: pricing},
                {name: 'コスト効率', score: cost}
            ];

            const strongest = scores.reduce((a, b) => a.score > b.score ? a : b);
            const weakest = scores.reduce((a, b) => a.score < b.score ? a : b);

            document.getElementById('strengthAnalysis').innerHTML = 
                `<p>${strongest.name}（${strongest.score.toFixed(1)}%）が最大の強み。この優位性を活かした戦略展開が重要。</p>`;

            document.getElementById('improvementAnalysis').innerHTML = 
                `<p>${weakest.name}（${weakest.score.toFixed(1)}%）の改善が優先課題。収益達成可能性向上の鍵となる領域。</p>`;

            // 推奨アクション更新
            let actions = [];
            if (trend < 75) actions.push('• オンライン教育プラットフォームと個別最適化システムの導入');
            if (brand < 80) actions.push('• 専門性アピールと受講生満足度向上施策の強化');
            if (content < 80) actions.push('• カリキュラム品質向上と学習成果の可視化');
            if (environment < 70) actions.push('• ハイブリッド教育体制とデジタル基盤の整備');
            if (pricing < 75) actions.push('• 料金体系の見直しと教育ROIの明確化');
            if (cost < 75) actions.push('• 講師配置最適化とコンテンツ制作効率化');

            if (actions.length === 0) {
                actions.push('• 現在の教育品質維持と更なる専門領域への展開');
                actions.push('• AI・VR等の新技術を活用した次世代教育サービスの開発');
            }

            document.getElementById('recommendedActions').innerHTML = 
                actions.map(action => `<div>${action}</div>`).join('');
        }

        // イベントリスナー設定
        document.addEventListener('DOMContentLoaded', function() {
            initializeChart();
            
            const sliders = document.querySelectorAll('input[type="range"]');
            sliders.forEach(slider => {
                // 初期値表示
                const valueElement = document.getElementById(slider.id + 'Value');
                if (valueElement) {
                    valueElement.textContent = slider.value + '%';
                }
                
                // スライダー変更時の処理
                slider.addEventListener('input', function() {
                    const valueElement = document.getElementById(this.id + 'Value');
                    if (valueElement) {
                        valueElement.textContent = this.value + '%';
                    }
                    updateAllScores();
                });
            });

            // 初期計算
            updateAllScores();
        });
